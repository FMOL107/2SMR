######################
# Interfaces de red  #
######################

#EDIT /etc/network/interfaces (configurar los par치metros est치ticos de la interfaz eth0: IP 192.168.5.240/24, gateway 192.168.5.1, servidores de nombres 192.168.5.1 y 8.8.8.8)
auto eth0
iface eth0 inet static
    address 192.168.5.240
    netmask 255.255.255.0
    gateway 192.168.5.1
    dns-nameservers 192.168.5.1 8.8.8.8

#Deshabilitar y habilitar la interfaz de red eth0, configurando los par치metros establecidos en /etc/network/interfaces
ifdown eth0
ifup eth0

##################
# Servicio proxy #
##################

#Instalar Squid
apt-get install squid

#Archivo de log por defecto de Squid
cat /var/log/squid

#EDIT /etc/squid/squid.conf (habilitar las peticiones procedentes de 192.168.0.0/16 en Squid)
acl localnet src 192.168.0.0/16
http_access allow localnet

#############
# Navegador #
#############

#Descargar la p치gina predeterminada del servidor es.wikipedia.org, utilizando 192.168.5.1 como proxy
export http_proxy=192.168.5.1:3128
export https_proxy=192.168.5.1:3128
wget es.wikipedia.org

#Establecer 192.168.5.1 como proxy de forma permanente para wget y el usuario actual
echo export http_proxy=192.168.5.1:3128 >> $HOME/.bashrc
echo export https_proxy=192.168.5.1:3128 >> $HOME/.bashrc
