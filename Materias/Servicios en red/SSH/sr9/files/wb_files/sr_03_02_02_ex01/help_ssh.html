#####################
# Interfaces de red #
#####################

#EDIT /etc/network/interfaces (configurar los parámetros estáticos de la interfaz eth0: IP 192.168.5.240/24, gateway 192.168.5.1, servidores de nombres 192.168.5.1 y 8.8.8.8)
auto eth0
iface eth0 inet static
    address 192.168.5.240
    netmask 255.255.255.0
    gateway 192.168.5.1
    dns-nameservers 192.168.5.1 8.8.8.8

#EDIT /etc/network/interfaces (configurar los parámetros de la interfaz eth0 utilizando un cliente DHCP)
auto eth0
iface eth0 inet dhcp

#Deshabilitar y habilitar la interfaz de red eth0, configurando los parámetros establecidos en /etc/network/interfaces
ifdown eth0
ifup eth0

################
# Servicio SSH #
################

#Instalar OpenSSH
apt-get install openssh-server

#EDIT /etc/ssh/sshd_config (habilitar la autenticación de clave pública)
RSAAuthentication yes
PubkeyAuthentication yes

#EDIT /etc/ssh/sshd_config (habilitar el reenvío X11)
X11Forwarding yes

#Crear usuario user1
adduser user1

#Mostrar quién está logueado local o remotamente
who

#Obtener listado de los últimos usuarios logueados
last

#Obtener la lista de los últimos usuarios logueados, incluyendo los intentos fallidos
lastb

#Obtener la fecha y hora de inicio de sesión más reciente de cada usuario
lastlog

###############
# Cliente SSH #
###############

#Generar un par de claves, por defecto en ~/.ssh/id_rsa.pub (pública) y ~/.ssh/id_rsa (privada)
ssh-key

#Transferir la clave pública, por defecto en ~/.ssh/id_rsa.pub, a un servidor SSH (dado el usuario user1 en el servidor host1); la clave pública se añadirá de forma predeterminada al archivo  ~/.ssh/authorized_keys del servidor, y ya no se requerirá contraseña para la autenticación
ssh-copy-id user1@host1

#Iniciar sesión desde un terminal remoto en el servidor host1 como usuario user1
ssh user1@host1

#Iniciar sesión desde un terminal remoto en el servidor host1 como usuario user1, habilitando el reenvío de mensajes X11 al cliente (el cliente debe ejecutar un servidor X)
ssh -X user1@host1

#Ejecutar el comando ls en el servidor host1 como usuario usuario1
ssh user1@host1 ls

#Iniciar sesión desde un terminal remoto en el servidor host1 como usuario user1, reenviando datos desde el puerto local 8888 al puerto remoto 80
ssh -L 127.0.0.1:8888:127.0.0.1:80 user1@host1
